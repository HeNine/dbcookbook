\begin{recipe}
[ %
	preparationtime = {\SI{15}{\minute}},
%	bakingtime={\SI{1}{\hour}},
%	bakingtemperature={\protect\bakingtemperature{topbottomheat=\SI{280}{\celsius}}},
	portion = {\portion{8}},
	source = {chrusher}
    ]{Salmon Rolls}

    \introduction{
    	I think this counts as authentic British Columbian cuisine -- smoked salmon was a common food in British Columbia long before white people showed up and the mix of ingredients from different cultures is on brand for the multicultural nature of Canadian cuisine. I want to say I got this recipe from my mum but I have no idea where she got it from.

    	An easy to make appetizer.

    	If made with wild pacific salmon and spinach tortillas it is quite colourful.
	}

	\ingredients[]{
		4  & tortillas (ideally spinach ones for the green colour)\\
		\SI{250}{\gram} & Cream cheese \\
		\SI{250}{\gram} & Smoked salmon -- sliced or trimmings
	}

	\preparation{
		\step Spread the cream cheese on one side of each of the tortillas.

		\step Distribute smoked salmon on top of cream cheese.

		\step Roll the tortillas such that the cream cheese and smoked salmon are on the inside.

		\step Slice the rolled tortillas into slices about \SI{7}{\milli\meter} thick.
	}


\end{recipe}
