\begin{recipe}
[ %
	preparationtime = {\SIrange{5}{8}{\hour}},
%	bakingtime={\SI{1}{\hour}},
%	bakingtemperature={\protect\bakingtemperature{topbottomheat=\SI{280}{\celsius}}},
	portion = {\portion{6}},
	source = {HPBraincase},
	sourceref = {\href{http://www.slowcookercentral.com/recipe/sweet-lamb-curry/}{Slow Cooker Central}}
    ]{Sweet Lamb Curry}


%    \introduction{
%
%	}

	\ingredients[]{
		\SI{1}{\kilo\gram} & Diced lamb \\
		\SI{1}{large} & Onion, diced \\
		2 & White yams or white sweet potatoes, diced into \SI{1}{\inch} chunks \\
		\SI{2}{cloves} & Garlic, minced \\
		\SI{3}{\teaspoon} & Mild curry powder \\
		\SI{1/2}{\teaspoon} & Ginger, minced \\
		\SI{2}{small} & Green granny smith apples, pealed, cored and diced \\
		\SI{1/2}{\cup}  & Beef stock liquid \\
		\SI{1/2}{\cup}  & Fruit chutney \\
		& Cornflour or all-purpose flour to thicken
	}

	\preparation{
		\step Place all ingredients (except cornflour) into slow cooker.

		\step Cook on HIGH for \SI{2}{\hour}, then LOW for \SI{3}{\hour} (or \SI{5}{\hour} on auto).

		If you wish it to cook all day, this dish can also be cooked longer on low heat.

		\step \SI{1/2}{\hour} prior to serving, mix \SIrange{2}{3}{\tablespoon} cornflour with \SI{1}{\tablespoon} water to form a paste and stir into the dish. Cook uncovered for the last \SI{15}{\minute} to thicken.

		\step Serve with creamy mashed potato or rice and crusty bread rolls.

	}

\end{recipe}
