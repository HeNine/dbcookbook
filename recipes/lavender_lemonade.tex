\begin{recipe}
[ %
	preparationtime = {\SI{20}{\minute}},
%	bakingtime={\SI{1}{\hour}},
%	bakingtemperature={\protect\bakingtemperature{topbottomheat=\SI{280}{\celsius}}},
	portion = {\portion{6}},
	source = {JRSkinnyCat},
	sourceref = {\href{https://www.simplyrecipes.com/recipes/lavender_lemonade/}{Simply Recipes}}
    ]{Lavender Lemonade}


    \introduction{
		The sugar/lemon balance will depend which type of lemons you use, (Meyer lemons are less tart than regular lemons), and your personal preference for sweetness.

		For the lavender, do not use florist flowers. If you use fresh garden flowers, make sure they are pesticide free.
	}

	\ingredients[13]{
		\SI{1}{\ounce} & Freshly picked and rinsed lavender flowers \textbf{OR} \\
		\SI{1}{\tablespoon} & Dried lavender flowers \\
		\SI{1}{\cup} & White, granulated sugar \\
		\SI{2}{cups} & Boiling water \\
		\SI{1x1/2}{\cup} & Freshly squeezed lemon juice \\
		\SI{2}{cups} or more & cold water \\
		& Ice
	}

	\preparation{
		\step Cut the lavender flowers from the stems and place in a medium bowl or large pyrex measuring bowl. Pour the sugar over the flowers, then gently rub the flowers into the sugar with your fingers.

		\step Pour \SI{2}{cups} of boiling water over the lavender sugar and stir until the sugar has melted. Cover and let infuse for \SI{30}{\minute} to several hours.

		\step Strain the lavender-infused simple syrup and pour into a serving carafe or pitcher.

		\step Stir in the lemon juice. Add another two cups of water. Taste and adjust for tartness. Add more lemon juice if too sweet. Add more sugar if too tart. Add ice and more water to desired level of concentration.

		Note that the ice will melt eventually, further diluting the drink.

		\step For presentation, add thin lemon slices and a few lavender sprigs to the serving pitcher.
	}

\end{recipe}
