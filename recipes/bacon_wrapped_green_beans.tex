\begin{recipe}
[ %
		preparationtime = {\SI{30}{\minute}},
		bakingtime={\SIrange{10}{15}{\minute}},
		bakingtemperature={\protect\bakingtemperature{topbottomheat=\SI{400}{\fahrenheit} (\SI{205}{\celsius})}},
		portion = {\portion{4}},
		source = {JRSkinnyCat},
		sourceref = {\href{https://www.tasteofhome.com/recipes/bacon-wrapped-green-beans/}{Taste of Home}}
    ]{Bacon-Wrapped Green Beans}


%    \introduction{
%
%	}

	\ingredients[9]{
		\SI{3/4}{\pound} & Fresh green beans \\
		4 & Bacon strips \\
		\SI{3}{\tablespoon} & Butter, melted \\
		\SI{1/4}{\cup}  & Packed brown sugar \\
		\SI{1/4}{\teaspoon} & Garlic salt \\
		\SI{1/8}{\teaspoon} & Soy sauce \\
		& Toothpicks, soaked in water
	}

	\preparation{
		\step Place beans in a large saucepan and cover with water. Bring to a boil. Cook, uncovered, for \SI{8}{\minute} or until crisp-tender.

		\step Meanwhile, in a skillet, cook bacon over medium heat until cooked but not crisp; about \SI{3}{\minute}.

		\step Remove bacon and place on paper towels to drain grease.

		\step Drain beans; place about 12 beans on each bacon strip. Wrap bacon around beans and secure with a toothpick. Place on an ungreased baking sheet.

		\step In a small bowl, combine the butter, brown sugar, garlic salt and soy sauce; brush mixture over bundles.

		\step Bake at \SI{400}{\fahrenheit} (\SI{205}{\celsius}) for \SIrange{10}{15}{\minute} or until bacon is crisp.
	}

\end{recipe}
